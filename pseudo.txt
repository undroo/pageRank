functions required


isContain(char *url, char *term){
	while(fscanf(url, input) != EOF){
		if(strcmp(input, term)){
			return TRUE;
		}
	}
}
	
	
frequency(char *url, char *term){
	int occurance = 0;
	while(fscanf(url, "%s", input) != EOF){
		if(strcmp(input, term)){
			occurance++;
		}
	}
	return occurance;
}
